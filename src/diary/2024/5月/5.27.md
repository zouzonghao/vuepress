```sh

  --name=syncthing \
  --hostname=5600Linux `#optional` \
  -p 8384:8384 \
  -p 22000:22000/tcp \
  -p 22000:22000/udp \
  -p 21027:21027/udp \
  -v /usr/docker/syncthing:/var/syncthing \
  -v /usr/blog:/var/syncthing/data/blog \
  -v /usr/note:/var/syncthing/data/note \
  --restart=always \
  syncthing/syncthing:latest
```
